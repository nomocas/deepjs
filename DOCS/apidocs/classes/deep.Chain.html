flatten

() deep.Chain chainable async

Provided by the deep module.

Defined in deep.js:1478

will perform FULL backgrounds application on chain entries. (see backgrounds documentation)

Success injected : entries values Errors injected : any flatten error

Returns:

deep.Chain: this

Example:

var a = {
    obj:{
        first:true
    },
    myFunc:function(){
        console.log("base myFunc");
        this.obj.a = true;
    }
}

var b = {
    backgrounds:[a],
    obj:{
        second:true
    },
    myFunc:deep.compose.after(function()
    {
        console.log("myFunc of b : ", this)
        this.obj.b = true;
    })
}

deep({})
.bottom(b)
.flatten()
.run("myFunc")
.query("./obj")
.equal({
      first:true,
      second:true,
   a:true,
   b:true
});

deep({
    sub:{
        backgrounds:[b],
        obj:{
            third:true
        }
    }
})
.flatten()
.query("/sub")
.run("myFunc")
.query("./obj")
.equal({
   first:true,
      second:true,
      third:true,
   a:true,
   b:true
});
